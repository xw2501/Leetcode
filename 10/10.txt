public class Solution {
    public boolean isMatch(String s, String p) {
        if(s.equals(p))             //¼òµ¥ÅÐ¶Ï
            return true;
        if(s==null && p==null)
            return true;
        if(s==null && p!=null)
            return false;
        if(s!=null && p==null)
            return false;
        
        int slen = s.length();
        int plen = p.length();
        
        if(slen==0 && plen==0)
            return true;
        if(slen!=0 && plen==0)
            return false;
        if(p.charAt(plen-1)=='.' && slen>0) {
            return isMatch(s.substring(0,slen-1), p.substring(0,plen-1));
        }
        else if(p.charAt(plen-1)=='*') {
            if(plen>1) {
                if(slen>0) {
                    if(p.charAt(plen-2)=='.') {
                        if(isMatch(s.substring(0,slen-1),p))
                            return true;
                    }
                    else if(p.charAt(plen-2)=='*')
                        return false;
                    else {
                        if(p.charAt(plen-2)==s.charAt(slen-1)) {
                            if(isMatch(s.substring(0,slen-1),p))
                                return true;
                        }
                    }
                }
                return isMatch(s,p.substring(0,plen-2));
            }
        }
        else {
            if(slen>0) {
                if(p.charAt(plen-1)==s.charAt(slen-1))
                    return isMatch(s.substring(0,slen-1),p.substring(0,plen-1));
            }
        }
        return false;
    }
}
